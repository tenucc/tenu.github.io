template <typename T>
Rank Vector<T>::insert(Rank r, T const & e){
    expand();
    for (int i = _size; i > r; i--){
        _elem[i] = _elem[i-1];   
    }
    _elem[r]=e; _size++;
    return r;
}

//区间删除
template <typename T>
int Vector<T>::remove(Rank lo, Rank hi){
    if ( lo == hi) return 0;
    while(hi < _size) _elem[lo++] ==  _elem[hi++];
    _size = lo; shrink();
    return hi - lo;
}

//单元素删除
template < typename T>
T Vector<T>::remove(Rank r){
    T e = _elem[r];
    remove(r,r+1);
    return e;
}

    while ((lo<hi--) && (e!=_elem[hi]));
    return hi;
}

    int n = 0;
